-- Таблица для хранения Happ ссылок
CREATE TABLE IF NOT EXISTS happ_links (
  id SERIAL PRIMARY KEY,
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  subscription_id INTEGER REFERENCES subscriptions(id),
  happ_url TEXT NOT NULL,
  provider_id TEXT NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  expires_at TIMESTAMP WITH TIME ZONE,
  is_active BOOLEAN DEFAULT true,
  UNIQUE(user_id, subscription_id)
);

-- Таблица для хранения устройств пользователей
CREATE TABLE IF NOT EXISTS user_devices (
  id SERIAL PRIMARY KEY,
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  device_hwid TEXT NOT NULL,
  device_name TEXT,
  device_type TEXT,
  last_seen TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  UNIQUE(user_id, device_hwid)
);

-- Индексы
CREATE INDEX IF NOT EXISTS idx_happ_links_user_id ON happ_links(user_id);
CREATE INDEX IF NOT EXISTS idx_happ_links_expires ON happ_links(expires_at);
CREATE INDEX IF NOT EXISTS idx_user_devices_user_id ON user_devices(user_id);
CREATE INDEX IF NOT EXISTS idx_user_devices_hwid ON user_devices(device_hwid);

-- Функция для подсчета устройств пользователя
CREATE OR REPLACE FUNCTION get_user_devices_count(p_user_id UUID)
RETURNS INTEGER
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = ''
AS $$
DECLARE
  device_count INTEGER;
BEGIN
  SELECT COUNT(*) INTO device_count
  FROM user_devices
  WHERE user_id = p_user_id;
  
  RETURN device_count;
END;
$$;

-- Функция для поиска по email с датой регистрации
CREATE OR REPLACE FUNCTION find_user_by_email(search_email TEXT)
RETURNS JSON
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = ''
AS $$
DECLARE
  result JSON;
BEGIN
  SELECT json_build_object(
    'id', id,
    'email', email,
    'created_at', created_at,
    'last_sign_in_at', last_sign_in_at
  ) INTO result
  FROM auth.users
  WHERE email = search_email
  LIMIT 1;
  
  RETURN result;
END;
$$;

-- Функция для получения всех пользователей с датами
CREATE OR REPLACE FUNCTION get_all_users()
RETURNS JSON
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = ''
AS $$
DECLARE
  result JSON;
BEGIN
  SELECT json_agg(
    json_build_object(
      'id', id,
      'email', email,
      'created_at', created_at,
      'last_sign_in_at', last_sign_in_at
    ) ORDER BY created_at DESC
  ) INTO result
  FROM auth.users;
  
  RETURN COALESCE(result, '[]'::JSON);
END;
$$;

-- Даем права
GRANT EXECUTE ON FUNCTION find_user_by_email TO anon, authenticated;
GRANT EXECUTE ON FUNCTION get_all_users TO anon, authenticated;
GRANT EXECUTE ON FUNCTION get_user_devices_count TO anon, authenticated;
