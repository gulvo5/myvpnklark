-- Создаем таблицы и функции (полный SQL скрипт)
CREATE TABLE IF NOT EXISTS user_devices (
  id SERIAL PRIMARY KEY,
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  device_hwid TEXT NOT NULL,
  device_name TEXT,
  device_type TEXT,
  last_seen TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  UNIQUE(user_id, device_hwid)
);

-- Включаем RLS
ALTER TABLE public.user_devices ENABLE ROW LEVEL SECURITY;

-- Политики RLS
DROP POLICY IF EXISTS "Users can view own devices" ON public.user_devices;
DROP POLICY IF EXISTS "Service role can insert devices" ON public.user_devices;
DROP POLICY IF EXISTS "Service role can update devices" ON public.user_devices;
DROP POLICY IF EXISTS "Service role can delete devices" ON public.user_devices;

CREATE POLICY "Users can view own devices" ON public.user_devices FOR SELECT TO authenticated USING (auth.uid() = user_id);
CREATE POLICY "Service role can insert devices" ON public.user_devices FOR INSERT TO service_role WITH CHECK (true);
CREATE POLICY "Service role can update devices" ON public.user_devices FOR UPDATE TO service_role USING (true) WITH CHECK (true);
CREATE POLICY "Service role can delete devices" ON public.user_devices FOR DELETE TO service_role USING (true);

GRANT ALL ON public.user_devices TO authenticated;
GRANT ALL ON public.user_devices TO service_role;

-- Создаем функцию для поиска по email
CREATE OR REPLACE FUNCTION find_user_by_email(search_email TEXT)
RETURNS JSON
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE result JSON;
BEGIN
  SELECT json_build_object('id', id, 'email', email, 'created_at', created_at) INTO result
  FROM auth.users WHERE email = search_email LIMIT 1;
  RETURN result;
END;
$$;

-- Создаем функцию для получения всех пользователей
CREATE OR REPLACE FUNCTION get_all_users()
RETURNS JSON
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE result JSON;
BEGIN
  SELECT json_agg(json_build_object('id', id, 'email', email, 'created_at', created_at)) INTO result FROM auth.users;
  RETURN COALESCE(result, '[]'::JSON);
END;
$$;

GRANT EXECUTE ON FUNCTION find_user_by_email TO anon, authenticated;
GRANT EXECUTE ON FUNCTION get_all_users TO anon, authenticated;
