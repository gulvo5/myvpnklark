-- Создаем функцию для поиска по email
CREATE OR REPLACE FUNCTION find_user_by_email(search_email TEXT)
RETURNS JSON
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = ''
AS $$
DECLARE
  result JSON;
BEGIN
  SELECT json_build_object(
    'id', id,
    'email', email
  ) INTO result
  FROM auth.users
  WHERE email = search_email
  LIMIT 1;
  
  RETURN result;
END;
$$;

-- Создаем функцию для получения всех пользователей
CREATE OR REPLACE FUNCTION get_all_users()
RETURNS JSON
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = ''
AS $$
DECLARE
  result JSON;
BEGIN
  SELECT json_agg(
    json_build_object(
      'id', id,
      'email', email
    )
  ) INTO result
  FROM auth.users;
  
  RETURN COALESCE(result, '[]'::JSON);
END;
$$;

-- Даем права
GRANT EXECUTE ON FUNCTION find_user_by_email TO anon, authenticated;
GRANT EXECUTE ON FUNCTION get_all_users TO anon, authenticated;
