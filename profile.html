async function adminGrantSubscription() { const identifierInput = document.getElementById('admin-identifier'); const dateInput = document.getElementById('admin-end-date'); const devicesInput = document.getElementById('admin-max-devices'); const messageEl = document.getElementById('admin-message'); const infoEl = document.getElementById('admin-info'); if (!identifierInput || !dateInput || !devicesInput || !messageEl) { console.error('Не найдены элементы админ-панели'); return; } const identifier = identifierInput.value.trim(); const endDateStr = dateInput.value.trim(); const maxDevices = parseInt(devicesInput.value) || 10; if (!identifier || !endDateStr) { messageEl.textContent = 'Заполните идентификатор и дату'; messageEl.style.color = 'red'; return; } let endsAt; try { const [dd, mm, yyyy] = endDateStr.split('.').map(p => p.trim()); if (!dd || !mm || !yyyy || yyyy.length !== 4) { throw new Error('неверный формат даты'); } endsAt = `${yyyy}-${mm.padStart(2, '0')}-${dd.padStart(2, '0')}`; const testDate = new Date(endsAt); if (Number.isNaN(testDate.getTime())) { throw new Error('некорректная дата'); } } catch { messageEl.textContent = 'Неверный формат. Пример: 31.12.2026'; messageEl.style.color = 'red'; return; } messageEl.textContent = 'Поиск пользователя...'; messageEl.style.color = 'yellow'; if (infoEl) { infoEl.classList.remove('hidden'); infoEl.innerHTML = `🔍 Поиск: "${identifier}" в таблице profiles...`; } try { // Сначала проверим, существует ли вообще таблица profiles if (infoEl) infoEl.innerHTML += `
🔍 Проверяем доступ к таблице profiles...`; // Пробуем получить список всех колонок в таблице const { data: columns, error: columnsError } = await supabaseClient .from('information_schema.columns') .select('column_name') .eq('table_name', 'profiles') .eq('table_schema', 'public'); if (columnsError) { if (infoEl) infoEl.innerHTML += `
❌ Не удалось получить структуру таблицы: ${columnsError.message}`; } else if (columns && columns.length > 0) { const columnNames = columns.map(c => c.column_name).join(', '); if (infoEl) infoEl.innerHTML += `
📋 Колонки в таблице: ${columnNames}`; } // Получаем все записи из таблицы для диагностики if (infoEl) infoEl.innerHTML += `
🔍 Загружаем все записи из profiles...`; const { data: allProfiles, error: allError } = await supabaseClient .from('profiles') .select('*') .limit(50); if (allError) { if (infoEl) infoEl.innerHTML += `
❌ Ошибка доступа к таблице profiles: ${allError.message}`; throw allError; } if (infoEl) { infoEl.innerHTML += `
📊 Всего записей в таблице: ${allProfiles?.length || 0}`; if (allProfiles && allProfiles.length > 0) { infoEl.innerHTML += `
📋 Первые 5 записей:`; allProfiles.slice(0, 5).forEach((profile, index) => { // Показываем все поля каждой записи const fields = Object.entries(profile) .map(([key, value]) => `${key}: "${value}"`) .join(', '); infoEl.innerHTML += `
${index + 1}. ${fields}`; }); // Проверяем, есть ли наш UUID в любой колонке const foundInAnyField = allProfiles.find(p => { return Object.values(p).some(val => String(val).toLowerCase().includes(identifier.toLowerCase()) ); }); if (foundInAnyField) { infoEl.innerHTML += `
✅ Найдено совпадение в какой-то колонке!`; } } } let profile = null; const isUuid = isUUID(identifier); if (isUuid) { if (infoEl) infoEl.innerHTML += `
🔍 Определен как UUID, ищем по всем возможным полям...`; // Список возможных полей, где может храниться UUID const possibleIdFields = ['id', 'user_id', 'uuid', 'userid', 'profile_id', 'auth_id']; for (const field of possibleIdFields) { if (infoEl) infoEl.innerHTML += `
📌 Пробуем поле "${field}": "${identifier}"`; const { data, error } = await supabaseClient .from('profiles') .select('*') .eq(field, identifier) .maybeSingle(); if (error && error.code !== 'PGRST116') { if (infoEl) infoEl.innerHTML += `
⚠️ Ошибка при поиске по ${field}: ${error.message}`; continue; } if (data) { profile = data; if (infoEl) infoEl.innerHTML += `
✅ Найдено по полю "${field}"!`; break; } } // Если не нашли по точному UUID, пробуем ilike if (!profile) { if (infoEl) infoEl.innerHTML += `
📌 Пробуем ILIKE поиск по всем полям...`; for (const field of possibleIdFields) { const { data, error } = await supabaseClient .from('profiles') .select('*') .ilike(field, `%${identifier}%`) .maybeSingle(); if (error) continue; if (data) { profile = data; if (infoEl) infoEl.innerHTML += `
✅ Найдено по ILIKE в поле "${field}"!`; break; } } } } else { if (infoEl) infoEl.innerHTML += `
🔍 Определен как username, начинаем поиск...`; // Вариант 1: точное совпадение if (infoEl) infoEl.innerHTML += `
📌 1. Точное совпадение в username: "${identifier}"`; let { data, error } = await supabaseClient .from('profiles') .select('*') .eq('username', identifier) .maybeSingle(); if (error && error.code !== 'PGRST116') throw error; profile = data; // Вариант 2: нижний регистр if (!profile) { const lowerIdentifier = identifier.toLowerCase(); if (infoEl) infoEl.innerHTML += `
📌 2. Нижний регистр: "${lowerIdentifier}"`; const { data, error } = await supabaseClient .from('profiles') .select('*') .eq('username', lowerIdentifier) .maybeSingle(); if (error && error.code !== 'PGRST116') throw error; profile = data; } // Вариант 3: ilike (регистронезависимый) if (!profile) { if (infoEl) infoEl.innerHTML += `
📌 3. ILIKE в username: "${identifier}"`; const { data, error } = await supabaseClient .from('profiles') .select('*') .ilike('username', identifier) .maybeSingle(); if (error && error.code !== 'PGRST116') throw error; profile = data; } // Вариант 4: ilike с wildcards if (!profile) { if (infoEl) infoEl.innerHTML += `
📌 4. ILIKE с % в username: "%${identifier}%"`; const { data, error } = await supabaseClient .from('profiles') .select('*') .ilike('username', `%${identifier}%`) .maybeSingle(); if (error && error.code !== 'PGRST116') throw error; profile = data; } // Вариант 5: поиск по email if (!profile) { if (infoEl) infoEl.innerHTML += `
📌 5. Поиск по email: "${identifier}@"`; const { data, error } = await supabaseClient .from('profiles') .select('*') .ilike('email', `${identifier}@%`) .maybeSingle(); if (error && error.code !== 'PGRST116') throw error; if (data) profile = data; } } if (!profile) { // Если всё еще не нашли, показываем все записи для ручного поиска if (infoEl && allProfiles && allProfiles.length > 0) { infoEl.innerHTML += `

🔍 Все доступные записи:`; allProfiles.forEach((p, idx) => { infoEl.innerHTML += `
${idx + 1}. ${JSON.stringify(p)}`; }); } messageEl.textContent = `❌ Пользователь с ${isUuid ? 'UUID' : 'username'} "${identifier}" не найден`; messageEl.style.color = 'red'; if (infoEl) infoEl.innerHTML += `
❌ Все варианты поиска не дали результатов`; return; } if (infoEl) { infoEl.innerHTML += `
✅ НАЙДЕН! Полные данные:`; infoEl.innerHTML += `
${JSON.stringify(profile, null, 2)}`; } messageEl.textContent = `Пользователь найден! Выдача подписки...`; const subscriptionData = { user_id: profile.id, status: 'active', ends_at: endsAt + 'T23:59:59Z', max_devices: maxDevices, used_devices: 0, updated_at: new Date().toISOString() }; if (infoEl) infoEl.innerHTML += `
📝 Данные подписки: ${JSON.stringify(subscriptionData)}`; const { error: subError } = await supabaseClient .from('subscriptions') .upsert(subscriptionData, { onConflict: 'user_id', ignoreDuplicates: false }); if (subError) throw subError; // Если всё успешно, показываем конфетти if (window.confetti) { confetti({ particleCount: 100, spread: 70, origin: { y: 0.6 } }); } messageEl.textContent = `✅ Подписка оформлена для пользователя "${profile.username || identifier}" до ${endDateStr} (макс. ${maxDevices} устройств)`; messageEl.style.color = 'lime'; showToast('Подписка успешно выдана!', 'success'); if (infoEl) infoEl.innerHTML += `
✅ Подписка успешно оформлена!`; identifierInput.value = ''; dateInput.value = ''; devicesInput.value = '10'; document.getElementById('clear-identifier')?.classList.add('hidden'); } catch (err) { console.error('Ошибка выдачи подписки:', err); messageEl.textContent = 'Ошибка: ' + (err.message || 'неизвестно'); messageEl.style.color = 'red'; showToast('Ошибка: ' + err.message, 'error'); if (infoEl) infoEl.innerHTML += `
❌ Ошибка: ${err.message}`; } }
